数据太大，long long存不下也没必要，采用字符串处理末位即可。

#include<iostream>
#include<cstring>
using namespace std;
int main()
{
    char s[110];
    int n;
    cin>>n;

for(int i=1;i<=n;i++)
{
    cin>>s;
    int l=strlen(s)-1;
    if((s[l]-'0')%2==0) cout<<"even";
    else cout<<"odd";

    cout<<endl;
}
return 0;

}

#include<iostream>
#include<algorithm>
#include<cmath>
using namespace std;
int a[10010];

bool IS_PRIME(int x)
{
    if(x<2) return false;
    for(int i=2;i<=sqrt(x);i++)
    {
        if(x%i==0) return false;

}
return true;

}
int main()
{
    int n,k;
    cin>>n>>k;

    for(int i=1;i<=n;i++)
    {
        cin>>a[i];
    }
    sort(a,a+n+1);
    int ans=a[n-k+1]-a[k];
    if(IS_PRIME(ans)) cout<<"YES";
    else cout<<"NO";
    cout<<endl<<ans<<endl;
    return 0;
}

判断方向与是否出界。

#include<cstdio>
#include<iostream>
#include<cstring>

using namespace std;

int pic[25][25];

    int dx[]={1,-1,0,1};
    int dy[]={0,1,1,-1};
int main()
{
    memset(pic,-1,sizeof(pic));
    int n;
    cin>>n;
    for(int i=1;i<=n;i++)
{
    for(int j=1;j<=n;j++)
    {
        pic[i][j]=0;
    }
}

int ix=1,iy=1;
int cnt=0;
for(int i=1;i<=n*n;i++)
{

    pic[ix][iy]=i;
    if(ix==n && iy==n) break;
    if(cnt==0 || cnt==2) {cnt=(cnt+1)%4;
    }
    while(pic[ ix+dx[cnt] ][ iy+dy[cnt] ]) cnt=(cnt+1)%4;
    ix=ix+dx[cnt];
    iy=iy+dy[cnt];
}
for(int my=1;my<=n;my++)
    for(int mx=1;mx<=n;mx++)
        {
            printf("%4d",pic[my][mx]);
            if(mx==n) cout<<"\n";

        }
return 0;
}
